class Ex{

class Bar{
private:

public:
    Bar():y(0){}

        if (y+a>100)  throw Ex(a*y);
        else {y=a; cout<<y<<endl;}
    }
};
class Robot{
private:
    int a[11][11]={{0},{0}};
    int x,y;
public:
    int getX(){return x;}
    int getY(){return y;}
    Robot():x(5),y(5){a[5][5]=1;};
    void move(int a,int b){
        if (((x+a)>=11 | (y+b)>=11)|((x+a)<0 | (y+b)<0)) throw "OffTheField";
        this->a[x][y]=0;
        x=x+a;
        y=y+b;
        this->a[x][y]=1;

    }
    void print(){
        static int count=0;
        for (int i=0;i<11;i++){
            for (int y=0;y<11;y++){
                if (a[i][y]==1){
                    cout<<" X ";
                    x=i;
                    this->y=y;
                }
                else cout<<"[ ]";
            }
            cout<<endl;
        }
        count++;
        cout<<"Ход "<<count<<endl;
    }
};
